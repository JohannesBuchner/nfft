00038 static void reconstruct(char* filename,int N,int M,int iteration , int weight)
00039 {
00040   int j,k,l;
00041   double time,min_time,max_time,min_inh,max_inh;
00042   ticks t0, t1;
00043   double t,real,imag;
00044   double w,epsilon=0.0000003;     /* epsilon is a the break criterium for
00045                                    the iteration */;
00046   mri_inh_2d1d_plan my_plan;
00047   solver_plan_complex my_iplan;
00048   FILE* fp,*fw,*fout_real,*fout_imag,*finh,*ftime;
00049   int my_N[3],my_n[3];
00050   int flags = PRE_PHI_HUT| PRE_PSI |MALLOC_X| MALLOC_F_HAT|
00051                       MALLOC_F| FFTW_INIT| FFT_OUT_OF_PLACE;
00052   unsigned infft_flags = CGNR | PRECOMPUTE_DAMP;
00053 
00054   double Ts;
00055   double W,T;
00056   int N3;
00057   int m=2;
00058   double sigma = 1.25;
00059 
00060   ftime=fopen("readout_time.dat","r");
00061   finh=fopen("inh.dat","r");
00062 
00063   min_time=INT_MAX; max_time=INT_MIN;
00064   for(j=0;j<M;j++)
00065   {
00066     fscanf(ftime,"%le ",&time);
00067     if(time<min_time)
00068       min_time = time;
00069     if(time>max_time)
00070       max_time = time;
00071   }
00072 
00073   fclose(ftime);
00074 
00075   Ts=(min_time+max_time)/2.0;
00076 
00077 
00078   min_inh=INT_MAX; max_inh=INT_MIN;
00079   for(j=0;j<N*N;j++)
00080   {
00081     fscanf(finh,"%le ",&w);
00082     if(w<min_inh)
00083       min_inh = w;
00084     if(w>max_inh)
00085       max_inh = w;
00086   }
00087   fclose(finh);
00088 
00089   N3=ceil((MAX(fabs(min_inh),fabs(max_inh))*(max_time-min_time)/2.0+(m)/(2*sigma))*4*sigma);
00090   /* N3 has to be even */
00091   if(N3%2!=0)
00092     N3++;
00093 
00094   T=((max_time-min_time)/2.0)/(0.5-((double) (m))/N3);
00095   W=N3/T;
00096 
00097   my_N[0]=N; my_n[0]=ceil(N*sigma);
00098   my_N[1]=N; my_n[1]=ceil(N*sigma);
00099   my_N[2]=N3; my_n[2]=N3;
00100 
00101   /* initialise nfft */
00102   mri_inh_2d1d_init_guru(&my_plan, my_N, M, my_n, m, sigma, flags,
00103                       FFTW_MEASURE| FFTW_DESTROY_INPUT);
00104 
00105 
00106   /* precompute lin psi if set */
00107   if(my_plan.plan.nfft_flags & PRE_LIN_PSI)
00108     nfft_precompute_lin_psi(&my_plan.plan);
00109 
00110   if (weight)
00111     infft_flags = infft_flags | PRECOMPUTE_WEIGHT;
00112 
00113   /* initialise my_iplan, advanced */
00114   solver_init_advanced_complex(&my_iplan,(nfft_mv_plan_complex*)(&my_plan), infft_flags );
00115 
00116   /* get the weights */
00117   if(my_iplan.flags & PRECOMPUTE_WEIGHT)
00118   {
00119     fw=fopen("weights.dat","r");
00120     for(j=0;j<my_plan.M_total;j++)
00121     {
00122         fscanf(fw,"%le ",&my_iplan.w[j]);
00123     }
00124     fclose(fw);
00125   }
00126 
00127   /* get the damping factors */
00128   if(my_iplan.flags & PRECOMPUTE_DAMP)
00129   {
00130     for(j=0;j<N;j++){
00131       for(k=0;k<N;k++) {
00132         int j2= j-N/2;
00133         int k2= k-N/2;
00134         double r=sqrt(j2*j2+k2*k2);
00135         if(r>(double) N/2)
00136           my_iplan.w_hat[j*N+k]=0.0;
00137         else
00138           my_iplan.w_hat[j*N+k]=1.0;
00139       }
00140     }
00141   }
00142 
00143   fp=fopen(filename,"r");
00144   ftime=fopen("readout_time.dat","r");
00145 
00146   for(j=0;j<my_plan.M_total;j++)
00147   {
00148     fscanf(fp,"%le %le %le %le",&my_plan.plan.x[2*j+0],&my_plan.plan.x[2*j+1],&real,&imag);
00149     my_iplan.y[j]=real+ _Complex_I*imag;
00150     fscanf(ftime,"%le ",&my_plan.t[j]);
00151 
00152     my_plan.t[j] = (my_plan.t[j]-Ts)/T;
00153   }
00154   fclose(fp);
00155   fclose(ftime);
00156 
00157 
00158   finh=fopen("inh.dat","r");
00159   for(j=0;j<N*N;j++)
00160   {
00161     fscanf(finh,"%le ",&my_plan.w[j]);
00162     my_plan.w[j]/=W;
00163   }
00164   fclose(finh);
00165 
00166 
00167   if(my_plan.plan.nfft_flags & PRE_PSI) {
00168     nfft_precompute_psi(&my_plan.plan);
00169   }
00170   if(my_plan.plan.nfft_flags & PRE_FULL_PSI) {
00171       nfft_precompute_full_psi(&my_plan.plan);
00172   }
00173 
00174   /* init some guess */
00175   for(j=0;j<my_plan.N_total;j++)
00176   {
00177     my_iplan.f_hat_iter[j]=0.0;
00178   }
00179 
00180   t0 = getticks();
00181 
00182   /* inverse trafo */
00183   solver_before_loop_complex(&my_iplan);
00184   for(l=0;l<iteration;l++)
00185   {
00186     /* break if dot_r_iter is smaller than epsilon*/
00187     if(my_iplan.dot_r_iter<epsilon)
00188     break;
00189     fprintf(stderr,"%e,  %i of %i\n",sqrt(my_iplan.dot_r_iter),
00190     l+1,iteration);
00191     solver_loop_one_step_complex(&my_iplan);
00192   }
00193 
00194   t1 = getticks();
00195   t = nfft_elapsed_seconds(t1,t0);
00196 
00197   fout_real=fopen("output_real.dat","w");
00198   fout_imag=fopen("output_imag.dat","w");
00199 
00200   for (j=0;j<N*N;j++) {
00201     /* Verschiebung wieder herausrechnen */
00202     my_iplan.f_hat_iter[j]*=cexp(-2.0*_Complex_I*KPI*Ts*my_plan.w[j]*W);
00203 
00204     fprintf(fout_real,"%le ",creal(my_iplan.f_hat_iter[j]));
00205     fprintf(fout_imag,"%le ",cimag(my_iplan.f_hat_iter[j]));
00206   }
00207 
00208   fclose(fout_real);
00209   fclose(fout_imag);
00210   solver_finalize_complex(&my_iplan);
00211   mri_inh_2d1d_finalize(&my_plan);
00212 }
