00033 static void accuracy_nsfft(int d, int J, int M, int m)

00041              sizeof(double _Complex));
00042   swap_sndft_adjoint=(double _Complex*) nfft_malloc(p.N_total*
00043                sizeof(double _Complex));
00044 
00045   nsfft_init_random_nodes_coeffs(&p);
00046 
00048   nsfft_trafo_direct(&p);

00050   CSWAP(swap_sndft_trafo,p.f);

00055   printf("%5d\t %+.5E\t",J,
00056          X(error_l_infty_1_complex)(swap_sndft_trafo, p.f, p.M_total,

00058   fflush(stdout);

00060   nfft_vrand_unit_complex(p.f, p.M_total);
00061 
00063   nsfft_adjoint_direct(&p);

00066

00071          X(error_l_infty_1_complex)(swap_sndft_adjoint, p.f_hat,

00073                                  p.f, p.M_total));

00075 
00076   nfft_free(swap_sndft_adjoint);
