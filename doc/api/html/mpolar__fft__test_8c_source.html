00062 static int mpolar_grid(int T, int S, double *x, double *w)
00063 {
00064   int t, r;
00065   double W;
00066   int R2=2*ceil(sqrt(2.0)*S/2);
00067   double xx, yy;
00068   int M=0;
00069 
00070   for(t=-T/2; t<T/2; t++)
00071   {
00072     for(r=-R2/2; r<R2/2; r++)
00073     {
00074       xx = (double)r/S*cos(KPI*t/T);
00075       yy = (double)r/S*sin(KPI*t/T);
00076 
00077       if ( ((-0.5-1.0/(double)S)<=xx) & (xx<=(0.5+1.0/(double)S)) &
00078         ((-0.5-1.0/(double)S)<=yy) & (yy<=(0.5+1.0/(double)S)) )
00079       {
00080         x[2*M+0] = xx;
00081         x[2*M+1] = yy;
00082 
00083         if (r==0)
00084           w[M] = 1.0/4.0;
00085         else
00086           w[M] = fabs((double)r);
00087 
00088         M++; 
00089       }
00090     }
00091   }
00092 
00094    W=0.0;
00095    for (t=0; t<M; t++)
00096       W+=w[t];
00097 
00098    for (t=0; t<M; t++)
00099     w[t]/=W;
00100 
00101   return M;                             
00102 }
