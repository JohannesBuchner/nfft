00042 {

00044   double _Complex *f_hat_k, *f_j;
00045   double _Complex exp_omega_0;
00046 
00047   for(j=0, f_j=ths->f; j<ths->M_total; j++, f_j++)
00048     (*f_j) =0;

00050   for(j=0, f_j=ths->f; j<ths->M_total; j++, f_j++)
