00278 static inline R evaluate_chebyshev(const INT n, const R *c, const R x)
00279 {
00280   R a = c[n-2], b = c[n-1], t;
00281   INT j;
00282   
00283   A(n >= 2);
00284   
00285   for (j = n - 2; j > 0; j--)
00286   {
00287     t = c[j-1] - b;
00288     b = a + K(2.0) * x * b;
00289     a = t;
00290   }
00291   return a + x * b;
00292 }
