00063 static inline int scaled_modified_bessel_i_series(const R x, const R alpha,
00064   const int nb, const int ize, R *b)
00065 {
00066   const R enmten = K(4.0)*nfft_float_property(NFFT_R_MIN);
00067   R tempa = K(1.0), empal = K(1.0) + alpha, halfx = K(0.0), tempb = K(0.0);
00068   int n, ncalc = nb;
00069 
00070   if (enmten < x)
00071     halfx = x/K(2.0);
00072 
00073   if (alpha != K(0.0))
00074     tempa = POW(halfx, alpha)/TGAMMA(empal);
00075 
00076   if (ize == 2)
00077     tempa *= EXP(-x);
00078 
00079   if (K(1.0) < x + K(1.0))
00080     tempb = halfx*halfx;
00081 
00082   b[0] = tempa + tempa*tempb/empal;
00083 
00084   if (x != K(0.0) && b[0] == K(0.0))
00085     ncalc = 0;
00086 
00087   if (nb == 1)
00088     return ncalc;
00089 
00090   if (K(0.0) < x)
00091   {
00092     R tempc = halfx, tover = (enmten + enmten)/x;
00093 
00094     if (tempb != K(0.0))
00095       tover = enmten/tempb;
00096 
00097     for (n = 1; n < nb; n++)
00098     {
00099       tempa /= empal;
00100       empal += K(1.0);
00101       tempa *= tempc;
00102 
00103       if (tempa <= tover*empal)
00104         tempa = K(0.0);
00105 
00106       b[n] = tempa + tempa*tempb/empal;
00107 
00108       if (b[n] == K(0.0) && n < ncalc)
00109         ncalc = n;
00110     }
00111   }
00112   else
00113     for (n = 1; n < nb; n++)
00114       b[n] = K(0.0);
00115 
00116   return ncalc;
00117 }

00119 static inline void scaled_modified_bessel_i_normalize(const R x,
00120   const R alpha, const int nb, const int ize, R *b, const R sum_)
00121 {
00122   const R enmten = K(4.0)*nfft_float_property(NFFT_R_MIN);
00123   R sum = sum_, tempa;
00124   int n;
00125 
00126   /* Normalize, i.e., divide all b[n] by sum */
00127   if (alpha != K(0.0))
00128     sum = sum * TGAMMA(K(1.0) + alpha) * POW(x/K(2.0), -alpha);
00129 
00130   if (ize == 1)
00131     sum *= EXP(-x);
00132 
00133   tempa = enmten;
00134 
00135   if (K(1.0) < sum)
00136     tempa *= sum;
00137 
00138   for (n = 1; n <= nb; n++)
00139   {
00140     if (b[n-1] < tempa)
00141       b[n-1] = K(0.0);
00142 
00143     b[n-1] /= sum;
00144   }
00145 }

00194 static int smbi(const R x, const R alpha, const int nb, const int ize, R *b)
00195 {
00196   /* machine dependent parameters */
00197   /* NSIG   - DECIMAL SIGNIFICANCE DESIRED.  SHOULD BE SET TO */
00198   /*          IFIX(ALOG10(2)*NBIT+1), WHERE NBIT IS THE NUMBER OF */
00199   /*          BITS IN THE MANTISSA OF A WORKING PRECISION VARIABLE. */
00200   /*          SETTING NSIG LOWER WILL RESULT IN DECREASED ACCURACY */
00201   /*          WHILE SETTING NSIG HIGHER WILL INCREASE CPU TIME */
00202   /*          WITHOUT INCREASING ACCURACY.  THE TRUNCATION ERROR */
00203   /*          IS LIMITED TO A RELATIVE ERROR OF T=.5*10**(-NSIG). */
00204   /* ENTEN  - 10.0 ** K, WHERE K IS THE LARGEST int SUCH THAT */
00205   /*          ENTEN IS MACHINE-REPRESENTABLE IN WORKING PRECISION. */
00206   /* ENSIG  - 10.0 ** NSIG. */
00207   /* RTNSIG - 10.0 ** (-K) FOR THE SMALLEST int K SUCH THAT */
00208   /*          K .GE. NSIG/4. */
00209   /* ENMTEN - THE SMALLEST ABS(X) SUCH THAT X/4 DOES NOT UNDERFLOW. */
00210   /* XLARGE - UPPER LIMIT ON THE MAGNITUDE OF X WHEN IZE=2.  BEAR */
00211   /*          IN MIND THAT IF ABS(X)=N, THEN AT LEAST N ITERATIONS */
00212   /*          OF THE BACKWARD RECURSION WILL BE EXECUTED. */
00213   /* EXPARG - LARGEST WORKING PRECISION ARGUMENT THAT THE LIBRARY */
00214   /*          EXP ROUTINE CAN HANDLE AND UPPER LIMIT ON THE */
00215   /*          MAGNITUDE OF X WHEN IZE=1. */
00216   const int nsig = MANT_DIG + 2;
00217   const R enten = nfft_float_property(NFFT_R_MAX);
00218   const R ensig = POW(K(10.0),(R)nsig);
00219   const R rtnsig = POW(K(10.0),-CEIL((R)nsig/K(4.0)));
00220   const R xlarge = K(1E4);
00221   const R exparg = FLOOR(LOG(POW(K(R_RADIX),K(DBL_MAX_EXP-1))));
00222 
00223   /* System generated locals */
00224   int l, n, nend, magx, nbmx, ncalc, nstart;
00225   R p, em, en, sum, pold, test, empal, tempa, tempb, tempc, psave, plast, tover,
00226     emp2al, psavel;
00227 
00228   magx = LRINT(FLOOR(x));
00229 
00230   /* return if x, nb, or ize out of range */
00231   if (   nb <= 0 || x < K(0.0) || alpha < K(0.0) || K(1.0) <= alpha
00232       || ((ize != 1 || exparg < x) && (ize != 2 || xlarge < x)))
00233     return (MIN(nb,0) - 1);
00234 
00235   /* 2-term ascending series for small x */
00236   if (x < rtnsig)
00237     return scaled_modified_bessel_i_series(x,alpha,nb,ize,b);
00238 
00239   ncalc = nb;
00240   /* forward sweep, Olver's p-sequence */
00241 
00242   nbmx = nb - magx;
00243   n = magx + 1;
00244 
00245   en = (R) (n+n) + (alpha+alpha);
00246   plast = K(1.0);
00247   p = en/x;
00248 
00249   /* significance test */
00250   test = ensig + ensig;
00251 
00252   if ((5*nsig) < (magx << 1))
00253     test = SQRT(test*p);
00254   else
00255     test /= POW(K(1.585),(R)magx);
00256 
00257   if (3 <= nbmx)
00258   {
00259     /* calculate p-sequence until n = nb-1 */
00260     tover = enten/ensig;
00261     nstart = magx+2;
00262     nend = nb - 1;
00263 
00264     for (n = nstart; n <= nend; n++)
00265     {
00266       en += K(2.0);
00267       pold = plast;
00268       plast = p;
00269       p = en*plast/x + pold;
00270       if (p > tover)
00271       {
00272         /* divide p-sequence by tover to avoid overflow. Calculate p-sequence
00273          * until 1 <= |p| */
00274         tover = enten;
00275         p /= tover;
00276         plast /= tover;
00277         psave = p;
00278         psavel = plast;
00279         nstart = n + 1;
00280 
00281         do
00282         {
00283           n++;
00284           en += K(2.0);
00285           pold = plast;
00286           plast = p;
00287           p = en*plast/x + pold;
00288         } while (p <= K(1.0));
00289 
00290         tempb = en/x;
00291 
00292         /* Backward test. Find ncalc as the largest n such that test is passed. */
00293         test = pold*plast*(K(0.5) - K(0.5)/(tempb * tempb))/ensig;
00294         p = plast*tover;
00295         n--;
00296         en -= K(2.0);
00297         nend = MIN(nb,n);
00298 
00299         for (ncalc = nstart; ncalc <= nend; ncalc++)
00300         {
00301           pold = psavel;
00302           psavel = psave;
00303           psave = en*psavel/x + pold;
00304           if (test < psave * psavel)
00305             break;
00306         }
00307 
00308         ncalc--;
00309         goto L80;
00310       }
00311     }
00312 
00313     n = nend;
00314     en = (R) (n+n) + (alpha+alpha);
00315 
00316     /* special significance test for 2 <= nbmx */
00317     test = FMAX(test,SQRT(plast*ensig)*SQRT(p+p));
00318   }
00319 
00320   /* calculate p-sequence until significance test is passed */
00321   do
00322   {
00323     n++;
00324     en += K(2.0);
00325     pold = plast;
00326     plast = p;
00327     p = en*plast/x + pold;
00328   } while (p < test);
00329 
00330   /* Initialize backward recursion and normalization sum. */
00331 L80:
00332   n++;
00333   en += K(2.0);
00334   tempb = K(0.0);
00335   tempa = K(1.0)/p;
00336   em = (R)(n-1);
00337   empal = em + alpha;
00338   emp2al = em - K(1.0) + (alpha+alpha);
00339   sum = tempa*empal*emp2al/em;
00340   nend = n-nb;
00341 
00342   if (nend < 0)
00343   {
00344     /* We have n <= nb. So store b[n] and set higher orders to zero */
00345     b[n-1] = tempa;
00346     nend = -nend;
00347     for (l = 1; l <= nend; ++l)
00348       b[n-1 + l] = K(0.0);
00349   }
00350   else
00351   {
00352     if (nend != 0)
00353     {
00354       /* recur backward via difference equation, calculating b[n] until n = nb */
00355       for (l = 1; l <= nend; ++l)
00356       {
00357         n--;
00358         en -= K(2.0);
00359         tempc = tempb;
00360         tempb = tempa;
00361         tempa = en*tempb/x + tempc;
00362         em -= K(1.0);
00363         emp2al -= K(1.0);
00364 
00365         if (n == 1)
00366           break;
00367 
00368         if (n == 2)
00369           emp2al = K(1.0);
00370 
00371         empal -= K(1.0);
00372         sum = (sum + tempa*empal)*emp2al/em;
00373       }
00374     }
00375 
00376     /* store b[nb] */
00377     b[n-1] = tempa;
00378 
00379     if (nb <= 1)
00380     {
00381       sum = sum + sum + tempa;
00382       scaled_modified_bessel_i_normalize(x,alpha,nb,ize,b,sum);
00383       return ncalc;
00384     }
00385 
00386     /* calculate and store b[nb-1] */
00387     n--;
00388     en -= 2.0;
00389     b[n-1] = en*tempa/x + tempb;
00390 
00391     if (n == 1)
00392     {
00393       sum = sum + sum + b[0];
00394       scaled_modified_bessel_i_normalize(x,alpha,nb,ize,b,sum);
00395       return ncalc;
00396     }
00397 
00398     em -= K(1.0);
00399     emp2al -= K(1.0);
00400 
00401     if (n == 2)
00402       emp2al = K(1.0);
00403 
00404     empal -= K(1.0);
00405     sum = (sum + b[n-1]*empal)*emp2al/em;
00406   }
00407 
00408   nend = n - 2;
00409 
00410   if (nend != 0)
00411   {
00412     /* Calculate and store b[n] until n = 2. */
00413     for (l = 1; l <= nend; ++l)
00414     {
00415       n--;
00416       en -= K(2.0);
00417       b[n-1] = en*b[n]/x + b[n+1];
00418       em -= K(1.0);
00419       emp2al -= K(1.0);
00420 
00421       if (n == 2)
00422         emp2al = K(1.0);
00423 
00424       empal -= K(1.0);
00425       sum = (sum + b[n-1]*empal)*emp2al/em;
00426     }
00427   }
00428 
00429   /* calculate b[1] */
00430   b[0] = K(2.0)*empal*b[1]/x + b[2];
00431   sum = sum + sum + b[0];
00432 
00433   scaled_modified_bessel_i_normalize(x,alpha,nb,ize,b,sum);
00434   return ncalc;
00435 }
