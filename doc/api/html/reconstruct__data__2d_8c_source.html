00040 static void reconstruct(char* filename,int N,int M,int iteration, int weight)
00041 {
00042   int j,k,l;                    /* some variables  */
00043   ticks t0, t1;
00044   double real,imag,t;           /* to read the real and imag part of a complex number */
00045   nfft_plan my_plan;            /* plan for the two dimensional nfft  */
00046   solver_plan_complex my_iplan; /* plan for the two dimensional infft */
00047   FILE* fin;                    /* input file                         */
00048   FILE* fout_real;              /* output file                        */
00049   FILE* fout_imag;              /* output file                        */
00050   int my_N[2],my_n[2];          /* to init the nfft */
00051   double epsilon=0.0000003;     /* epsilon is a the break criterium for
00052                                    the iteration */
00053   unsigned infft_flags = CGNR | PRECOMPUTE_DAMP;  /* flags for the infft*/
00054   int m = 6;
00055   double alpha = 2.0;
00056   /* initialise my_plan */
00057   my_N[0]=N; my_n[0]=ceil(N*alpha);
00058   my_N[1]=N; my_n[1]=ceil(N*alpha);
00059   nfft_init_guru(&my_plan, 2, my_N, M, my_n, m, PRE_PHI_HUT| PRE_PSI|
00060                          MALLOC_X| MALLOC_F_HAT| MALLOC_F|
00061                          FFTW_INIT| FFT_OUT_OF_PLACE,
00062                          FFTW_MEASURE| FFTW_DESTROY_INPUT);
00063 
00064   /* precompute lin psi if set */
00065   if(my_plan.nfft_flags & PRE_LIN_PSI)
00066     nfft_precompute_lin_psi(&my_plan);
00067 
00068   /* set the flags for the infft*/
00069   if (weight)
00070     infft_flags = infft_flags | PRECOMPUTE_WEIGHT;
00071 
00072   /* initialise my_iplan, advanced */
00073   solver_init_advanced_complex(&my_iplan,(nfft_mv_plan_complex*)&my_plan, infft_flags );
00074 
00075   /* get the weights */
00076   if(my_iplan.flags & PRECOMPUTE_WEIGHT)
00077   {
00078     fin=fopen("weights.dat","r");
00079     for(j=0;j<my_plan.M_total;j++)
00080     {
00081         fscanf(fin,"%le ",&my_iplan.w[j]);
00082     }
00083     fclose(fin);
00084   }
00085 
00086   /* get the damping factors */
00087   if(my_iplan.flags & PRECOMPUTE_DAMP)
00088   {
00089     for(j=0;j<N;j++){
00090       for(k=0;k<N;k++) {
00091         int j2= j-N/2;
00092         int k2= k-N/2;
00093         double r=sqrt(j2*j2+k2*k2);
00094         if(r>(double) N/2)
00095           my_iplan.w_hat[j*N+k]=0.0;
00096         else
00097           my_iplan.w_hat[j*N+k]=1.0;
00098       }
00099     }
00100   }
00101 
00102   /* open the input file */
00103   fin=fopen(filename,"r");
00104 
00105   /* read x,y,freal and fimag from the knots */
00106   for(j=0;j<my_plan.M_total;j++)
00107   {
00108     fscanf(fin,"%le %le %le %le ",&my_plan.x[2*j+0],&my_plan.x[2*j+1],
00109     &real,&imag);
00110     my_iplan.y[j] = real + _Complex_I*imag;
00111   }
00112 
00113   fclose(fin);
00114 
00115   /* precompute psi */
00116   if(my_plan.nfft_flags & PRE_PSI)
00117     nfft_precompute_psi(&my_plan);
00118 
00119   /* precompute full psi */
00120   if(my_plan.nfft_flags & PRE_FULL_PSI)
00121       nfft_precompute_full_psi(&my_plan);
00122 
00123   /* init some guess */
00124   for(k=0;k<my_plan.N_total;k++)
00125     my_iplan.f_hat_iter[k]=0.0;
00126 
00127   t0 = getticks();
00128 
00129   /* inverse trafo */
00130   solver_before_loop_complex(&my_iplan);
00131   for(l=0;l<iteration;l++)
00132   {
00133     /* break if dot_r_iter is smaller than epsilon*/
00134     if(my_iplan.dot_r_iter<epsilon)
00135       break;
00136     fprintf(stderr,"%e,  %i of %i\n",sqrt(my_iplan.dot_r_iter),
00137     l+1,iteration);
00138     solver_loop_one_step_complex(&my_iplan);
00139   }
00140 
00141 
00142   t1 = getticks();
00143   t=nfft_elapsed_seconds(t1,t0);
00144 
00145   fout_real=fopen("output_real.dat","w");
00146   fout_imag=fopen("output_imag.dat","w");
00147 
00148   for(k=0;k<my_plan.N_total;k++) {
00149     fprintf(fout_real,"%le ", creal(my_iplan.f_hat_iter[k]));
00150     fprintf(fout_imag,"%le ", cimag(my_iplan.f_hat_iter[k]));
00151   }
00152 
00153   fclose(fout_real);
00154   fclose(fout_imag);
00155 
00156   /* finalize the infft */
00157   solver_finalize_complex(&my_iplan);
00158 
00159   /* finalize the nfft */
00160   nfft_finalize(&my_plan);
00161 }
