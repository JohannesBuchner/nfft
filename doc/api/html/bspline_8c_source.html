00023 static inline void bspline_help(const INT k, const R x, R *scratch, const INT j,
00024   const INT ug, const INT og, const INT r)
00025 {
00026   INT i; /* Row index of the de Boor scheme */
00027   INT idx; /* Index in scratch */
00028   R a; /* Alpha of de Boor scheme */
00029 
00030   /* computation of one column */
00031   for (i = og + r - k + 1, idx = og; idx >= ug; i--, idx--)
00032   {
00033     a = (x - (R)i) / ((R)(k - j));
00034     scratch[idx] = (K(1.0) - a) * scratch[idx - 1] + a * scratch[idx];
00035   }
00036 }

00038 /* Evaluate the cardinal B-Spline B_{n-1} supported on [0,n]. */
00039 R X(bspline)(const INT k, const R _x, R *scratch)
00040 {
00041   const R kk = (R)k;
00042   R result_value;
00043   INT r;
00044   INT g1, g2; /* boundaries */
00045   INT j, idx, ug, og; /* indices */
00046   R a; /* Alpha of de Boor scheme*/
00047   R x = _x;
00048 
00049   result_value = K(0.0);
00050 
00051   if (K(0.0) < x && x < kk)
00052   {
00053     /* Exploit symmetry around k/2, maybe. */
00054     if ( (kk - x) < x)
00055     {
00056       x = kk - x;
00057     }
00058 
00059     r = (INT)LRINT(CEIL(x) - K(1.0));
00060 
00061     /* Explicit case for first interval. */
00062     if (r == 0)
00063     {
00064       result_value = K(1.0);
00065       for (j = 0; j < k - 1; j++)
00066         result_value *= x/((R)(j+1));
00067       return result_value;
00068     }
00069 
00070 
00071     for (idx = 0; idx < k; idx++)
00072       scratch[idx] = K(0.0);
00073 
00074     scratch[k-r-1] = K(1.0);
00075 
00076     /* Bounds of the algorithm. */
00077     g1 = r;
00078     g2 = k - 1 - r;
00079     ug = g2;
00080 
00081     /* g1 <= g2 */
00082 
00083     for (j = 1, og = g2 + 1; j <= g1; j++, og++)
00084     {
00085       a = (x + (R)(k - r - og - 1)) / ((R)(k - j));
00086       scratch[og] = (K(1.0) - a) * scratch[og-1];
00087       bspline_help(k, x, scratch, j, ug + 1, og - 1, r);
00088       a = (x + (R)(k - r - ug - 1)) / ((R)(k - j));
00089       scratch[ug] = a * scratch[ug];
00090     }
00091 
00092     for (og-- ; j <= g2; j++)
00093     {
00094       bspline_help(k, x, scratch, j, ug + 1, og, r);
00095       a = (x + (R)(k - r - ug - 1)) / ((R)(k - j));
00096       scratch[ug] = a * scratch[ug];
00097     }
00098 
00099     for(; j < k; j++)
00100     {
00101       ug++;
00102       bspline_help(k, x, scratch, j, ug, og, r);
00103     }
00104 
00105     result_value = scratch[k-1];
00106   }
00107 
00108   return(result_value);
00109 }
