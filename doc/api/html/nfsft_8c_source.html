00629   double stheta;       /*< Current angle theta for Clenshaw algorithm        */
00630   double sphi;         /*< Current angle phi for Clenshaw algorithm          */
00631 
00632 #ifdef MEASURE_TIME
00633   plan->MEASURE_TIME_t[0] = 0.0;
00634   plan->MEASURE_TIME_t[1] = 0.0;
00635   plan->MEASURE_TIME_t[2] = 0.0;
00636 #endif
00637 
00638   if (wisdom.flags & NFSFT_NO_DIRECT_ALGORITHM)
00639   {
00640     return;
00641   }
00642 
00643   /* Copy spherical Fourier coefficients, if necessary. */
00644   if (plan->flags & NFSFT_PRESERVE_F_HAT)
00645   {
00646     memcpy(plan->f_hat_intern,plan->f_hat,plan->N_total*
00647            sizeof(double _Complex));
00648   }
00649   else
00650   {
00651     plan->f_hat_intern = plan->f_hat;
00652   }
00653 
00654   /* Check, if we compute with L^2-normalized spherical harmonics. If so,
00655    * multiply spherical Fourier coefficients with corresponding normalization
00656    * weight. */
00657   if (plan->flags & NFSFT_NORMALIZED)
00658   {
00659     /* Traverse Fourier coefficients array. */
00660     #pragma omp parallel for default(shared) private(k,n)
00661     for (k = 0; k <= plan->N; k++)
00662     {
00663       for (n = -k; n <= k; n++)
00664       {
00665         /* Multiply with normalization weight. */
00666         plan->f_hat_intern[NFSFT_INDEX(k,n,plan)] *=
00667           sqrt((2*k+1)/(4.0*KPI));
00668       }
00669     }
00670   }
00671 
00672   /* Distinguish by bandwidth M. */
00673   if (plan->N == 0)
00674   {
00675     /* N = 0 */
00676 
00677     /* Constant function */
00678     for (m = 0; m < plan->M_total; m++)
00679     {
00680       plan->f[m] = plan->f_hat_intern[NFSFT_INDEX(0,0,plan)];
00681     }
00682   }
00683   else
00684   {
00685     /* N > 0 */
00686 
00687     /* Evaluate
00688      *   \sum_{k=0}^N \sum_{n=-k}^k a_k^n P_k^{|n|}(cos theta_m) e^{i n phi_m}
00689      *   = \sum_{n=-N}^N \sum_{k=|n|}^N a_k^n P_k^{|n|}(cos theta_m)
00690      *     e^{i n phi_m}.
00691      */
00692     #pragma omp parallel for default(shared) private(m,stheta,sphi,f_m,n,a,n_abs,alpha,gamma,it2,it1,k,temp)
00693     for (m = 0; m < plan->M_total; m++)
00694     {
00695       /* Scale angle theta from [0,1/2] to [0,pi] and apply cosine. */
00696       stheta = cos(2.0*KPI*plan->x[2*m+1]);
00697       /* Scale angle phi from [-1/2,1/2] to [-pi,pi]. */
00698       sphi = 2.0*KPI*plan->x[2*m];
00699 
00700       /* Initialize result for current node. */
00701       f_m = 0.0;
00702 
00703       /* For n = -N,...,N, evaluate
00704        *   b_n := \sum_{k=|n|}^N a_k^n P_k^{|n|}(cos theta_m)
00705        * using Clenshaw's algorithm.
00706        */
00707       for (n = -plan->N; n <= plan->N; n++)
00708       {
00709         /* Get pointer to Fourier coefficients vector for current order n. */
00710         a = &(plan->f_hat_intern[NFSFT_INDEX(0,n,plan)]);
00711 
00712         /* Take absolute value of n. */
00713         n_abs = abs(n);
00714 
00715         /* Get pointers to three-term recurrence coefficients arrays. */
00716         alpha = &(wisdom.alpha[ROW(n_abs)]);
00717         gamma = &(wisdom.gamma[ROW(n_abs)]);
00718 
00719         /* Clenshaw's algorithm */
00720         it2 = a[plan->N];
00721         it1 = a[plan->N-1];
00722         for (k = plan->N; k > n_abs + 1; k--)
00723         {
00724           temp = a[k-2] + it2 * gamma[k];
00725           it2 = it1 + it2 * alpha[k] * stheta;
00726           it1 = temp;
00727         }
00728 
00729         /* Compute final step if neccesary. */
00730         if (n_abs < plan->N)
00731         {
00732           it2 = it1 + it2 * wisdom.alpha[ROWK(n_abs)+1] * stheta;
00733         }
00734 
00735         /* Compute final result by multiplying the fixed part
00736          *   gamma_|n| (1-cos^2(theta))^{|n|/2}
00737          * for order n and the exponential part
00738          *   e^{i n phi}
00739          * and add the result to f_m.
00740          */
00741         f_m += it2 * wisdom.gamma[ROW(n_abs)] *
00742           pow(1- stheta * stheta, 0.5*n_abs) * cexp(_Complex_I*n*sphi);
00743       }
00744 
00745       /* Write result f_m for current node to array f. */
00746       plan->f[m] = f_m;
00747     }
00748   }
00749 }

00751 void nfsft_adjoint_direct(nfsft_plan *plan)
00752 {
00753   int m;               /*< The node index                                    */
00754   int k;               /*< The degree k                                      */
00755   int n;               /*< The order n                                       */
00756   int n_abs;           /*< The absolute value of the order n, ie n_abs = |n| */
00757   double *alpha;       /*< Pointer to current three-term recurrence
00758                            coefficient alpha_k^n for associated Legendre
00759                            functions P_k^n                                   */
00760   double *gamma;       /*< Pointer to current three-term recurrence
00761                            coefficient beta_k^n for associated Legendre
00762                            functions P_k^n                                   */
00763   double _Complex it1;  /*< Auxilliary variable for Clenshaw algorithm        */
00764   double _Complex it2;  /*< Auxilliary variable for Clenshaw algorithm        */
00765   double _Complex temp; /*< Auxilliary variable for Clenshaw algorithm        */
00766   double stheta;       /*< Current angle theta for Clenshaw algorithm        */
00767   double sphi;         /*< Current angle phi for Clenshaw algorithm          */
00768 
00769 #ifdef MEASURE_TIME
00770   plan->MEASURE_TIME_t[0] = 0.0;
00771   plan->MEASURE_TIME_t[1] = 0.0;
00772   plan->MEASURE_TIME_t[2] = 0.0;
00773 #endif
00774 
00775   if (wisdom.flags & NFSFT_NO_DIRECT_ALGORITHM)
00776   {
00777     return;
00778   }
00779 
00780   /* Initialise spherical Fourier coefficients array with zeros. */
00781   memset(plan->f_hat,0U,plan->N_total*sizeof(double _Complex));
00782 
00783   /* Distinguish by bandwidth N. */
00784   if (plan->N == 0)
00785   {
00786     /* N == 0 */
00787 
00788     /* Constant function */
00789     for (m = 0; m < plan->M_total; m++)
00790     {
00791       plan->f_hat[NFSFT_INDEX(0,0,plan)] += plan->f[m];
00792     }
00793   }
00794   else
00795   {
00796     /* N > 0 */
00797 
00798 #ifdef _OPENMP
00799       /* Traverse all orders n. */
00800       #pragma omp parallel for default(shared) private(n,n_abs,alpha,gamma,m,stheta,sphi,it2,it1,k,temp)
00801       for (n = -plan->N; n <= plan->N; n++)
00802       {
00803         /* Take absolute value of n. */
00804         n_abs = abs(n);
00805 
00806         /* Get pointers to three-term recurrence coefficients arrays. */
00807         alpha = &(wisdom.alpha[ROW(n_abs)]);
00808         gamma = &(wisdom.gamma[ROW(n_abs)]);
00809 
00810         /* Traverse all nodes. */
00811         for (m = 0; m < plan->M_total; m++)
00812         {
00813           /* Scale angle theta from [0,1/2] to [0,pi] and apply cosine. */
00814           stheta = cos(2.0*KPI*plan->x[2*m+1]);
00815           /* Scale angle phi from [-1/2,1/2] to [-pi,pi]. */
00816           sphi = 2.0*KPI*plan->x[2*m];
00817 
00818           /* Transposed Clenshaw algorithm */
00819 
00820           /* Initial step */
00821           it1 = plan->f[m] * wisdom.gamma[ROW(n_abs)] *
00822             pow(1 - stheta * stheta, 0.5*n_abs) * cexp(-_Complex_I*n*sphi);
00823           plan->f_hat[NFSFT_INDEX(n_abs,n,plan)] += it1;
00824           it2 = 0.0;
00825 
00826           if (n_abs < plan->N)
00827           {
00828             it2 = it1 * wisdom.alpha[ROWK(n_abs)+1] * stheta;
00829             plan->f_hat[NFSFT_INDEX(n_abs+1,n,plan)] += it2;
00830           }
00831 
00832           /* Loop for transposed Clenshaw algorithm */
00833           for (k = n_abs+2; k <= plan->N; k++)
00834           {
00835             temp = it2;
00836             it2 = alpha[k] * stheta * it2 + gamma[k] * it1;
00837             it1 = temp;
00838             plan->f_hat[NFSFT_INDEX(k,n,plan)] += it2;
00839           }
00840         }
00841       }
00842 #else
00843     /* Traverse all nodes. */
00844     for (m = 0; m < plan->M_total; m++)
00845     {
00846       /* Scale angle theta from [0,1/2] to [0,pi] and apply cosine. */
00847       stheta = cos(2.0*KPI*plan->x[2*m+1]);
00848       /* Scale angle phi from [-1/2,1/2] to [-pi,pi]. */
00849       sphi = 2.0*KPI*plan->x[2*m];
00850 
00851       /* Traverse all orders n. */
00852       for (n = -plan->N; n <= plan->N; n++)
00853       {
00854         /* Take absolute value of n. */
00855         n_abs = abs(n);
00856 
00857         /* Get pointers to three-term recurrence coefficients arrays. */
00858         alpha = &(wisdom.alpha[ROW(n_abs)]);
00859         gamma = &(wisdom.gamma[ROW(n_abs)]);
00860 
00861         /* Transposed Clenshaw algorithm */
00862 
00863         /* Initial step */
00864         it1 = plan->f[m] * wisdom.gamma[ROW(n_abs)] *
00865           pow(1 - stheta * stheta, 0.5*n_abs) * cexp(-_Complex_I*n*sphi);
00866         plan->f_hat[NFSFT_INDEX(n_abs,n,plan)] += it1;
00867         it2 = 0.0;
00868 
00869         if (n_abs < plan->N)
00870         {
00871           it2 = it1 * wisdom.alpha[ROWK(n_abs)+1] * stheta;
00872           plan->f_hat[NFSFT_INDEX(n_abs+1,n,plan)] += it2;
00873         }
00874 
00875         /* Loop for transposed Clenshaw algorithm */
00876         for (k = n_abs+2; k <= plan->N; k++)
00877         {
00878           temp = it2;
00879           it2 = alpha[k] * stheta * it2 + gamma[k] * it1;
00880           it1 = temp;
00881           plan->f_hat[NFSFT_INDEX(k,n,plan)] += it2;
00882         }
00883       }
00884     }
00885 #endif
00886   }
00887 
00888   /* Check, if we compute with L^2-normalized spherical harmonics. If so,
00889    * multiply spherical Fourier coefficients with corresponding normalization
00890    * weight. */
00891   if (plan->flags & NFSFT_NORMALIZED)
00892   {
00893     /* Traverse Fourier coefficients array. */
00894     #pragma omp parallel for default(shared) private(k,n)
00895     for (k = 0; k <= plan->N; k++)
00896     {
00897       for (n = -k; n <= k; n++)
00898       {
00899         /* Multiply with normalization weight. */
00900         plan->f_hat[NFSFT_INDEX(k,n,plan)] *=
00901           sqrt((2*k+1)/(4.0*KPI));
00902       }
00903     }
00904   }
00905 
00906   /* Set unused coefficients to zero. */
00907   if (plan->flags & NFSFT_ZERO_F_HAT)
00908   {
00909     for (n = -plan->N; n <= plan->N+1; n++)
00910     {
00911       memset(&plan->f_hat[NFSFT_INDEX(-plan->N-1,n,plan)],0U,
00912         (plan->N+1+abs(n))*sizeof(double _Complex));
00913     }
00914   }
00915 }

01204       //fprintf(stderr,"nfsft_adjoint: fpt_transposed\n");
01205       /* Use transposed FPT. */
01206       for (n = -plan->N; n <= plan->N; n++)
01207       {
01208         //fprintf(stderr,"nfsft_adjoint: Executing fpt_transposed\n");
01209         //fflush(stderr);
01210         fpt_transposed(wisdom.set,abs(n),
01211           &plan->f_hat[NFSFT_INDEX(abs(n),n,plan)],
01212           &plan->f_hat[NFSFT_INDEX(0,n,plan)],
01213           plan->N,0U);
01214       }
01215 #endif
01216     }
01217 #ifdef MEASURE_TIME
01218     t1 = getticks();
01219     plan->MEASURE_TIME_t[0] = nfft_elapsed_seconds(t1,t0);
01220 #endif
01221 
01222     /* Check, if we compute with L^2-normalized spherical harmonics. If so,
01223      * multiply spherical Fourier coefficients with corresponding normalization
01224      * weight. */
01225     if (plan->flags & NFSFT_NORMALIZED)
01226     {
01227       //fprintf(stderr,"nfsft_adjoint: Normalizing\n");
01228       //fflush(stderr);
01229       /* Traverse Fourier coefficients array. */
01230       #pragma omp parallel for default(shared) private(k,n)
01231       for (k = 0; k <= plan->N; k++)
01232       {
01233         for (n = -k; n <= k; n++)
01234         {
01235           /* Multiply with normalization weight. */
01236           plan->f_hat[NFSFT_INDEX(k,n,plan)] *=
01237             sqrt((2*k+1)/(4.0*KPI));
01238         }
01239       }
01240     }
01241 
01242     /* Set unused coefficients to zero. */
01243     if (plan->flags & NFSFT_ZERO_F_HAT)
01244     {
01245       //fprintf(stderr,"nfsft_adjoint: Setting to zero\n");
01246       //fflush(stderr);
01247       for (n = -plan->N; n <= plan->N+1; n++)
01248       {
01249         memset(&plan->f_hat[NFSFT_INDEX(-plan->N-1,n,plan)],0U,
01250           (plan->N+1+abs(n))*sizeof(double _Complex));
01251       }
01252     }
01253     //fprintf(stderr,"nfsft_adjoint: Finished\n");
01254     //fflush(stderr);
01255   }
