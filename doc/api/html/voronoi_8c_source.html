00024 void X(voronoi_weights_1d)(R *w, R *x, const INT M)
00025 {
00026   INT j;
00027 
00028   w[0] = (x[1]-x[0])/K(2.0);
00029 
00030   for(j = 1; j < M-1; j++)
00031     w[j] = (x[j+1]-x[j-1])/K(2.0);
00032 
00033   w[M-1] = (x[M-1]-x[M-2])/K(2.0);
00034 }

00036 void X(voronoi_weights_S2)(R *w, R *xi, INT M)
00037 {
00038   R *x;
00039   R *y;
00040   R *z;
00041   int j;
00042   int k;
00043   int el;
00044   int Mlocal = M;
00045   int lnew;

00060   R *vr;
00061   int lp;
00062   int lpl;
00063   int kv;
00064   R a;
00065 
00066   /* Allocate memory for auxilliary arrays. */
00067   x = (R*)X(malloc)(M * sizeof(R));
00068   y = (R*)X(malloc)(M * sizeof(R));
00069   z = (R*)X(malloc)(M * sizeof(R));
00070 
00071   list = (int*)X(malloc)((6*M-12+1)*sizeof(int));
00072   lptr = (int*)X(malloc)((6*M-12+1)*sizeof(int));
00073   lend = (int*)X(malloc)((M+1)*sizeof(int));
00074   near = (int*)X(malloc)((M+1)*sizeof(int));
00075   next = (int*)X(malloc)((M+1)*sizeof(int));
00076   dist = (R*)X(malloc)((M+1)*sizeof(R));
00077   ltri = (int*)X(malloc)((6*M+1)*sizeof(int));
00078   listc = (int*)X(malloc)((6*M-12+1)*sizeof(int));
00079   xc = (R*)X(malloc)((2*M-4+1)*sizeof(R));
00080   yc = (R*)X(malloc)((2*M-4+1)*sizeof(R));
00081   zc = (R*)X(malloc)((2*M-4+1)*sizeof(R));
00082   rc = (R*)X(malloc)((2*M-4+1)*sizeof(R));
00083   vr = (R*)X(malloc)(3*(2*M-4+1)*sizeof(R));
00084 
00085   /* Convert from spherical Coordinates in [0,1/2]x[-1/2,1/2) to Cartesian
00086    * coordinates. */
00087   for (k = 0; k < M; k++)
00088   {
00089     x[k] = SIN(K2PI*xi[2*k+1])*COS(K2PI*xi[2*k]);
00090     y[k] = SIN(K2PI*xi[2*k+1])*SIN(K2PI*xi[2*k]);
00091     z[k] = COS(K2PI*xi[2*k+1]);
00092   }
00093 
00094   /* Generate Delaunay triangulation. */
00095   trmesh_(&Mlocal, x, y, z, list, lptr, lend, &lnew, near, next, dist, &ier);
00096 
00097   /* Check error flag. */
00098   if (ier == 0)
00099   {
00100     /* Get Voronoi vertices. */
00101     crlist_(&Mlocal, &Mlocal, x, y, z, list, lend, lptr, &lnew, ltri, listc, &nb, xc,
00102       yc, zc, rc, &ier);
00103 
00104     if (ier == 0)
00105     {
00106       /* Calcuate sizes of Voronoi regions. */
00107       for (k = 0; k < M; k++)
00108       {
00109         /* Get last neighbour index. */
00110         lpl = lend[k];
00111         lp = lpl;
00112 
00113         j = 0;
00114         vr[3*j] = x[k];
00115         vr[3*j+1] = y[k];
00116         vr[3*j+2] = z[k];
00117 
00118         do
00119         {
00120           j++;
00121           /* Get next neighbour. */
00122           lp = lptr[lp-1];
00123           kv = listc[lp-1];
00124           vr[3*j] = xc[kv-1];
00125           vr[3*j+1] = yc[kv-1];
00126           vr[3*j+2] = zc[kv-1];
00127           /* fprintf(stderr, "lp = %ld\t", lp); */
00128         } while (lp != lpl);
00129 
00130         a = 0;
00131         for (el = 0; el < j; el++)
00132         {
00133           a += areas_(vr, &vr[3*(el+1)],&vr[3*(((el+1)%j)+1)]);
00134         }
00135 
00136         w[k] = a;
00137       }
00138     }
00139   }
00140 
00141   /* Deallocate memory. */
00142   X(free)(x);
00143   X(free)(y);
00144   X(free)(z);
