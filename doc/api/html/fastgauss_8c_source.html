00107 {
00108   int j,k,l;
00109

00111     ths->f[j] = 0;
00112 
00113   if(ths->flags & DGT_PRE_CEXP)
00114     for(j=0,l=0; j<ths->M; j++)
00115       for(k=0; k<ths->N; k++,l++)
00116         ths->f[j] += ths->alpha[k]*ths->pre_cexp[l];
