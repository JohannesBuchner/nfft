00060 static int polar_grid(int T, int S, double *x, double *w)
00061 {
00062   int t, r;
00063   double W=(double)T*(((double)S/2.0)*((double)S/2.0)+1.0/4.0);
00064 
00065   for(t=-T/2; t<T/2; t++)
00066   {
00067     for(r=-S/2; r<S/2; r++)
00068     {
00069       x[2*((t+T/2)*S+(r+S/2))+0] = (double)r/S*cos(KPI*t/T);
